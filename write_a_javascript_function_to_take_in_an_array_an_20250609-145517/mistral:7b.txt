 Here is a simple JavaScript function that takes an array as input and returns HTML markup for an unordered list (ul) containing each item:

```javascript
function arrayToHtmlList(arr) {
    let html = '<ul>';

    arr.forEach(item => {
        html += `<li>${item}</li>`;
    });

    html += '</ul>';
    return html;
}
```

You can use this function like so:

```javascript
let myArray = ['apple', 'banana', 'cherry'];
console.log(arrayToHtmlList(myArray)); // Outputs '<ul><li>apple</li><li>banana</li><li>cherry</li></ul>'
```
