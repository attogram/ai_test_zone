```bash
#!/bin/bash

function array_to_html_list() {
  # Convert the array to a string, using the IFS (Internal Field Separator) variable
  array_string=$( IFS=', ' -r "${1[@]}" )

  # Use the echo command to print the HTML list
  echo "<ul>"
  echo "<li>${array_string}</li>"
  echo "</ul>"
}

# Call the function with the array as an argument
array_to_html_list "your_array_here"
```
