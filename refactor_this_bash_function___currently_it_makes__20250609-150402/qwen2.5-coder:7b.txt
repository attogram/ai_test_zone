```bash
function addModelToIndexFile {
    responseWords=$(wc -w < "$modelFile" | awk '{print $1}')
    responseBytes=$(wc -c < "$modelFile" | awk '{print $1}')

    (
        echo "<tr><td class='left'><a href='./$model.html'>$model</a></td><td >$responseWords</td><td>${responseBytes}</td>"
        while read -r line; do
            value=$(echo "$line" | cut -d ':' -f2)
            if [[ -n "$value" ]]; then
                echo "<td>${value}</td>";
            fi
        done < "$statsFile"
        echo "</tr>"
    ) >> "$indexFile"
}
```

In this refactored version, the `while` loop reads each line of `$statsFile` directly and processes it to extract the value. This value is then checked for non-emptyness and added to the output if it's not empty. This way, you avoid using an intermediate array and achieve the same result more efficiently.
